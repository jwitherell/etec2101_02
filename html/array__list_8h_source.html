272 ArrayList<T>& operator= (const ArrayList<T>& other)
273 {
274 // Save away all data we need from other
275 T* other_data_copy = new T[other.mCapacity];
276 for (unsigned int i = 0; i < other.mSize; i++)
277 other_data_copy[i] = other[i];
278
279 unsigned int other_size = other.size();
280 unsigned int other_capacity = other.capacity();
281

283 // was why it was important to save away the data above before doing this.
284 clear();
285
286 // Now just assign values from the temporaries we saved away
287 mData = other_data_copy;

292 // b = this_array_list = a;

472 ArrayListIterator find(const T& val, const ArrayListIterator& start) const
473 {
474 if (start.mArrayList != this)
475 throw std::runtime_error("iterator must be based on this ArrayList");
476
477 // Make a copy of the ArrayListIterator we were given and one to hold the end iterator value
478 ArrayListIterator temp = start;

593 T remove(unsigned int index, bool resize_if_necessary = true)
594 {
595 if (index >= mSize || index < 0)
596 throw std::out_of_range("Invalid index: " + std::to_string(index));
597
598 // Get the value we'll return at the end (the element removed)
599 T result = mData[index];
600
601 // Move all elements that come after index down one spot
602 for (unsigned int i = index; i < mSize - 1; i++)
603 mData[i] = mData[i + 1];
604
605 // Decrement our size
606 mSize--;
607
608 // Shrink, if applicable and requested
609 if (resize_if_necessary)
610 shrink();
611
612 // Return the result
613 return result;

645 {

647 unsigned int num_removed = 0;
648 while (cur_index >= 0 && (unsigned int)cur_index < mSize)
649 {

651 cur_index = find(val, cur_index);

708 void grow()
709 {
710 if (mSize == mCapacity)
711 {
712 // Allocate what will become the new array
713 T* new_array = nullptr;
714 if (mCapacity == 0)
715 new_array = new T[msMinCapacity];
716 else
717 new_array = new T[mCapacity * 2];
718
719 // Copy over data from the old array (if any)
720 if (mData != nullptr)
721 {
722 for (unsigned int i = 0; i < mSize; i++)
723 new_array[i] = mData[i];
724
725 // Destroy the old array
726 delete[] mData;
727 }
728
729 // Make the new array *the* array
730 mData = new_array;
731
732 // Note that our capacity is now double what it used to be
733 if (mCapacity == 0)
734 mCapacity = msMinCapacity;
735 else
736 mCapacity *= 2;
737 }
